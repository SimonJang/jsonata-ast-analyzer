<objective>
Fix filter predicate path leak into HOF lambda parameter bindings. Create extractBasePaths() helper that separates collection-identity paths from filter predicate paths at the HOF call site, so only base paths are bound to lambda parameters while predicate paths are still emitted as legitimate data reads.

Purpose: The 4 FILT bugs all share the same root cause -- walkHigherOrderCall binds ALL paths from walkNode(dataArg) to lambda parameters, including filter predicate paths that should only appear as output paths, not as binding paths. This causes spurious paths like items.active.name when $v.name is accessed.

Output: Fixed walkHigherOrderCall, 2 new helper functions, 4 unskipped BUG tests, 10+ new regression tests
</objective>

function walkHigherOrderCall(
  node: FunctionNode,
  semantics: Record<number, string>,
  scope: ScopeTracker,
): string[] {
  const args = node.arguments;
  const paths: string[] = [];
  for (const arg of args) {
    if (arg.type !== "lambda") {
      paths.push(...walkNode(arg, scope));  // emits ALL paths including predicates
    }
  }
  const lambdaArg = args.find((a) => a.type === "lambda") as LambdaNode | undefined;
  if (lambdaArg) {
    const dataArg = args.find((a) => a.type !== "lambda");
    const dataArgPaths = dataArg ? walkNode(dataArg, scope) : [];  // BUG: binds predicate paths too
    paths.push(...walkLambdaWithBindings(lambdaArg, dataArgPaths, semantics, scope));
  }
  return paths;
}
