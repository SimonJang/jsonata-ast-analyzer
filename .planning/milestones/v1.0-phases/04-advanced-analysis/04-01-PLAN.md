```typescript
export function buildPathString(steps: AstNode[]): string | null {
  const segments: string[] = [];
  for (const step of steps) {
    switch (step.type) {
      case "name":   segments.push((step as NameNode).value); break;
      case "wildcard": segments.push("*"); break;
      case "descendant": segments.push("**"); break;
      // ADD: case "parent": segments.push("%"); break;
      default: break; // skip filter/sort stages
    }
  }
  return segments.length > 0 ? segments.join(".") : null;
}
```
